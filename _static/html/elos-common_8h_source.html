184#define crinitElosTryExec(session, sessionLock, func, err_msg, ...) \
185 __extension__({ \
186 int res = SAFU_RESULT_OK; \
187 \
188 if ((errno = pthread_mutex_lock(sessionLock)) != 0) { \
189 crinitErrnoPrint("Failed to lock elos session."); \
190 res = -1; \
191 } else { \
192 if ((session) != NULL && !(session)->connected) { \
193 free(session); \
194 (session) = NULL; \
195 } \
196 size_t retryCount = 0; \
197 while ((session) == NULL) { \
198 if (crinitElosGetVTable()->elosServer == NULL) { \
199 crinitErrPrint("Elos server configuration missing or not loaded yet."); \
200 res = -1; \
201 break; \
202 } else { \
203 res = crinitElosGetVTable()->connect(crinitElosGetVTable()->elosServer, \
204 crinitElosGetVTable()->elosPort, \
205 (crinitElosSession_t **)&(session)); \
206 if (res != SAFU_RESULT_OK) { \
207 crinitDbgInfoPrint("Failed to connect to elosd on %s:%d.", crinitElosGetVTable()->elosServer, \
208 crinitElosGetVTable()->elosPort); \
209 if (retryCount >= CRINIT_ELOS_CONNECTION_RETRIES) { \
210 crinitErrPrint("Maximum connection retries with elosd on %s:%d exceeded.", \
211 crinitElosGetVTable()->elosServer, crinitElosGetVTable()->elosPort); \
212 break; \
213 } \
214 usleep(CRINIT_ELOS_CONNECTION_RETRY_INTERVAL_US); \
215 retryCount++; \
216 } \
217 } \
218 } \
219 \
220 if (res == SAFU_RESULT_OK) { \
221 res = func(__VA_ARGS__); \
222 if (res != SAFU_RESULT_OK) { \
223 crinitErrPrint(err_msg); \
224 } \
225 \
226 if ((errno = pthread_mutex_unlock(sessionLock)) != 0) { \
227 crinitErrnoPrint("Failed to unlock elos session."); \
228 res = -1; \
229 } \
230 } \
231 } \
232 \
233 res; \
234 })
